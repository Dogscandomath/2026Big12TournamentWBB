def simulate_game(team_a, team_b):
    srs_a = team_a['srs']
    srs_b = team_b['srs']
    
    # Defaults
    margin = srs_a - srs_b
    
    # Overrides based on User Inputs
    
    # 1. Baylor vs OSU (Baylor +8.0)
    # Reason: Road dominance in prior matchup
    if (team_a['team'] == 'Baylor' and team_b['team'] == 'Oklahoma St.'):
        margin = 8.0 

    # 2. Baylor vs WVU (Baylor +4.5)
    # Reason: Transitive property via Utah (Baylor +23 vs WVU relative to Utah)
    if (team_a['team'] == 'Baylor' and team_b['team'] == 'West Virginia'):
        margin = 4.5

    # 3. Baylor vs Tech (Tech +3.5)
    # Reason: Tech won at Baylor
    if (team_a['team'] == 'Baylor' and team_b['team'] == 'Texas Tech'):
        margin = -3.5 # Tech favored
        
    # 4. WVU vs Tech (Tech +6.0)
    # Reason: Tech won at WVU
    if (team_a['team'] == 'West Virginia' and team_b['team'] == 'Texas Tech'):
        margin = -6.0 # Tech favored

    # Win Probability Calculation
    prob_a = 0.5 * (1 + np.math.erf(margin / (10.3 * np.sqrt(2))))
    
    if np.random.rand() < prob_a:
        return team_a
    else:
        return team_b
